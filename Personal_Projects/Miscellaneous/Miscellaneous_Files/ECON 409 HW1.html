### Out-of-sample, one-month ahead forecasts w/ distinct per-forecast 90% confidence interval

import warnings
from statsmodels.tools.sm_exceptions import ConvergenceWarning
warnings.simplefilter('ignore', ConvergenceWarning)

def ARIMA_Model(train, test, arima_order):
    ### Passing dataframe to ARIMA fitting outputs identical prediction values
    ### due to grabbing column name rather than column values; how does it even 
    ### get the first prediction from a string?
    History = [x for x in train]
    
    ## Generate predictions
    Predictions = list()
    Y_Hat_CI = np.zeros((len(test),2))
    
    for x in range(len(test)):
        Model = ARIMA(History, order=arima_order).fit()
        Y_Hat = Model.forecast()[0]
        Conf_Int = Model.get_forecast().conf_int(alpha = 0.1)
        Y_Hat_CI[x][0] = Conf_Int[0][0]
        Y_Hat_CI[x][1] = Conf_Int[0][1]
        #Y_Hat_CI.append(Conf_Int)
        Predictions.append(Y_Hat)
        History.append(test[x])
        
    Y_Hat_CI = np.hsplit(Y_Hat_CI, 2)
    
    test = pd.DataFrame(test)
    test['Predictions'] = Predictions
    test['Lower CI'] = Y_Hat_CI[0]
    test['Upper CI'] = Y_Hat_CI[1]
    
    return test

Y = ARIMA_Model(X_Train['Japan'], X_Test['Japan'], arima_order = (2,1,0))

### DMW Test
### V_hat
error_R = change
error_T = error
MSE_R = (np.sum(np.square(change)))/P
MSE_T = (np.sum(np.square(error)))/P
tmp = np.square(error_R) - np.square(error_T) - (MSE_R - MSE_T)
V_hat = np.sum(np.square(tmp))/P

### Statistic
DMW = (MSE_R - MSE_T)/np.sqrt(V_hat/P)

print('Since the DMW statitsic is equal to ' + str(DMW) + ',' + ' which is less than the critical value (' + str(round(norm.ppf(.95),2)) + '),')
print('we fail to reject the null hypothesis that the MP model does not outperform the random walk model.')

Since the DMW statitsic is equal to -1.0003509794550451, which is less than the critical value (1.64),
we fail to reject the null hypothesis that the MP model does not outperform the random walk model.
